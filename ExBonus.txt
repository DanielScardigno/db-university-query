Ex 1

select count(`id`), year(`enrolment_date`)
from `students`
group by year(`enrolment_date`)
order by year(`enrolment_date`);


Ex 2

select count(`id`), `office_address`
from `teachers`
group by `office_address`;


Ex 3

select `student_id`, avg(`vote`)
from `exam_student`
group by `student_id`;


Ex 4

from `degrees`
group by `department_id`;
